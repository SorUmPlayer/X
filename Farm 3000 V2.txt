-- [[ CONFIGURAÇÕES E TIMERS ]]
task.wait(5) 

local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")
local VirtualInputManager = game:GetService("VirtualInputManager")
local ProximityPromptService = game:GetService("ProximityPromptService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local TextChatService = game:GetService("TextChatService")
local RunService = game:GetService("RunService")
local SoundService = game:GetService("SoundService")

local player = Players.LocalPlayer or Players.PlayerAdded:Wait()

-- Coordenadas iniciais
local localSpawnaBoss = Vector3.new(430, -137, 95)

-- [[ FUNÇÃO PARA DELETAR TODOS OS SPIKES ]]
local function limparSpikes()
    local powerAnimations = Workspace:FindFirstChild("World") 
        and Workspace.World:FindFirstChild("PowerAnimations")
        
    if powerAnimations then
        for _, objeto in pairs(powerAnimations:GetChildren()) do
            if objeto.Name == "Spike" then
                objeto:Destroy()
            end
        end
    end
end

-- Deleta os spikes existentes logo no início
limparSpikes()

-- [[ SEGREDO: MONITOR DE CHAT PARA A MÚSICA ]]
local function tocarMusicaSecreta()
    local som = Instance.new("Sound")
    som.SoundId = "rbxassetid://92524084286516"
    som.Volume = 1
    som.Looped = false
    som.Parent = SoundService
    som:Play()
    som.Ended:Connect(function()
        som:Destroy()
    end)
end

if TextChatService and TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
    TextChatService.MessageReceived:Connect(function(textMessage)
        if string.lower(textMessage.Text) == "sor" then
            tocarMusicaSecreta()
        end
    end)
else
    ReplicatedStorage:WaitForChild("DefaultChatSystemChatEvents"):WaitForChild("OnMessageDoneFiltering").OnClientEvent:Connect(function(messageData)
        if messageData and string.lower(messageData.Message) == "sor" then
            tocarMusicaSecreta()
        end
    end)
end

-- [[ 1. SCRIPT DE AUTO COLETA ]]
local AutoCollectEnabled = true
local MaxDistance = 15

task.spawn(function()
    ProximityPromptService.PromptShown:Connect(function(prompt)
        if not AutoCollectEnabled then return end
        local character = player.Character
        if character and character:FindFirstChild("HumanoidRootPart") then
            local promptParent = prompt:FindFirstAncestorOfClass("Part") or prompt:FindFirstAncestorOfClass("MeshPart")
            if promptParent then
                local distance = (character.HumanoidRootPart.Position - promptParent.Position).Magnitude
                if distance <= MaxDistance then
                    prompt.HoldDuration = 0
                    prompt:InputHoldBegin()
                    task.wait()
                    prompt:InputHoldEnd()
                end
            else
                prompt.HoldDuration = 0
                prompt:InputHoldBegin()
                task.wait()
                prompt:InputHoldEnd()
            end
        end
    end)
end)

-- [[ FUNÇÕES AUXILIARES ]]
local function teleportar(posicao)
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        player.Character.HumanoidRootPart.CFrame = CFrame.new(posicao)
    end
end

-- TREENTELEPORT POR OBJETIVO + NOCLIP DINÂMICO
local function puxarAteObjetivo(velocidade)
    local char = player.Character
    if not char then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local objetivo = Workspace:FindFirstChild("World")
        and Workspace.World:FindFirstChild("Map")
        and Workspace.World.Map:FindFirstChild("Milkya_Temple")
        and Workspace.World.Map.Milkya_Temple:FindFirstChild("Map_Part")

    if not objetivo then 
        warn("Map_Part não foi encontrado no workspace ainda!")
        return 
    end

    local bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.MaxForce = Vector3.new(1e7, 1e7, 1e7)
    bodyVelocity.Velocity = Vector3.new(0, 0, 0)
    bodyVelocity.Parent = hrp

    local noclipConnection
    noclipConnection = RunService.Stepped:Connect(function()
        if char then
            for _, part in pairs(char:GetDescendants()) do
                if part:IsA("BasePart") then 
                    part.CanCollide = false 
                end
            end
        end
    end)

    while objetivo and objetivo.Parent == Workspace.World.Map.Milkya_Temple do
        local posicaoFinal = objetivo.Position
        local direcao = (posicaoFinal - hrp.Position).Unit
        bodyVelocity.Velocity = direcao * velocidade
        
        if (hrp.Position - posicaoFinal).Magnitude <= 5 then
            hrp.CFrame = CFrame.new(posicaoFinal)
        end
        
        task.wait()
        objetivo = Workspace.World.Map.Milkya_Temple:FindFirstChild("Map_Part")
    end

    if noclipConnection then noclipConnection:Disconnect() end
    
    bodyVelocity.Velocity = Vector3.new(0, 0, 0)
    hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
    hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    bodyVelocity:Destroy()

    for _, part in pairs(char:GetDescendants()) do
        if part:IsA("BasePart") then 
            part.CanCollide = true 
        end
    end
end

local function simularTecla(tecla)
    VirtualInputManager:SendKeyEvent(true, Enum.KeyCode[tecla:upper()], false, game)
    task.wait(0.1)
    VirtualInputManager:SendKeyEvent(false, Enum.KeyCode[tecla:upper()], false, game)
end

-- [[ EXECUÇÃO ]]
teleportar(localSpawnaBoss)

local boss = nil
repeat
    if Workspace:FindFirstChild("Enemys") then
        local enemyFolder = Workspace.Enemys:FindFirstChild("Double Fire Salamander")
        if enemyFolder and enemyFolder:FindFirstChild("Character") then
            boss = enemyFolder.Character
        end
    end
    task.wait(0.5)
until boss and boss:FindFirstChild("HumanoidRootPart")

if TextChatService and TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
    local channel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
    if channel then channel:SendAsync("wildmutt") end
else
    ReplicatedStorage.DefaultChatSystemChatEvents.SayMessageRequest:FireServer("wildmutt", "All")
end

task.wait(5)

-- Combate
simularTecla("R")
task.wait(0.2)

local bossHumanoid = boss:FindFirstChildOfClass("Humanoid")
while boss and boss:FindFirstChild("HumanoidRootPart") and (not bossHumanoid or bossHumanoid.Health > 0) do
    if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
        player.Character.HumanoidRootPart.CFrame = boss.HumanoidRootPart.CFrame * CFrame.new(0, 5, 0)
    end
    local habilidades = {"Z", "X", "C", "V", "F"}
    for _, tecla in ipairs(habilidades) do
        simularTecla(tecla)
        task.wait(0.1)
    end
    task.wait(0.2)
end

-- [[ COLETA FINAL ]]
task.wait(0.5)

-- Executa a limpeza dos Spikes novamente logo antes de voar para garantir que o caminho do Map_Part esteja 100% limpo
limparSpikes()

-- Ativa o treenteleport ultra rápido em direção ao objetivo limpo
puxarAteObjetivo(350) 

task.wait(2)
